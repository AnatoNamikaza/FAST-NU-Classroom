#Q-1
def replacer(a):
  ch = a[0]
  a = a.replace(ch,'@')
  a[0] = ch
  return a
s = 'Ooogle'
print(replacer(s.lower()))

A = ['xyz', 'apple', 'mix', 'xanadu', 'Max','aardvark']
B, C = [], []
[B.append(i) for i in A if i[0] == 'm' or i[0] == 'M']
[C.append(i) for i in A if i[0] != 'm' and i[0] != 'M']
print(B)
print(C)

A = sorted(C)
[B.append(a) for a in A]
print(B)
